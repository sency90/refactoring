#include <string>
#include "result.cpp"
#include "Status.h"

class ALU
{
	constexpr static int NA = -1;
	constexpr static int ERR_RESULT = 65535;
	enum { ADD,SUB,MUL };

	int operand1 = NA;
	int operand2 = NA;
	std::string OPCODE = "";

public:
	void setOperand1(int operand1) {
		this->operand1 = operand1;
	}

	void setOperand2(int operand2) {
		this->operand2 = operand2;
	}

	void setOPCODE(std::string OPCODE) {
		this->OPCODE = OPCODE;
	}

	void enableSignal(Result *r) {
		if(operand1 == NA) {
			r->setStatus(Status::OPERAND1_WRONG);
			return;
		}
		else if(operand2 == NA) {
			r->setStatus(Status::OPERAND2_WRONG);
			return;
		}

		int result;
		if(OPCODE == "ADD") result = operand1 + operand2;
		else if(OPCODE == "SUB") result = operand1 - operand2;
		else if(OPCODE == "MUL") result = operand1 * operand2;
		else {
			r->setResultOnlyIfNormalStatus(ERR_RESULT);
			r->setStatus(Status::OPCODE_WRONG);
			return;
		}
		r->setResultOnlyIfNormalStatus(result);
	}
};
